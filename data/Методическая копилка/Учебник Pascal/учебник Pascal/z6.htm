    Var a,b,c,d : Integer;
    Function f ( Var b : Integer; c : Integer): Integer;
    Var d : Integer;
    Begin
        a:=2; b:=b+1; d:=3; c:=b-a;
        Writeln(a,b,c,d); f:=d;
    End;
    Begin
        a:=0; b:=0; c:=0; d:=0;
        d:=f(a,b); Writeln(a,b,c,d);
    End.
